## 2. 根据用户获取最新的推文
```python
# 1. 通过用户名获取用户对象
screen_name = 'elonmusk'  # 用户名（不带@符号）
user = await client.get_user_by_screen_name(screen_name)

# 2. 获取该用户的最新推文
# 类型：'Tweets'（推文）, 'Replies'（回复）, 'Media'（媒体）, 'Likes'（点赞）
user_tweets = await user.get_tweets('Tweets', count=10)

# 3. 遍历推文
for tweet in user_tweets:
    print(f"内容: {tweet.full_text}")
    print(f"时间: {tweet.created_at_datetime}")
    print(f"点赞: {tweet.favorite_count}")
    print(f"链接: https://x.com/{tweet.user.screen_name}/status/{tweet.id}")

# 获取第一页
tweets = await user.get_tweets('Tweets', count=20)

# 获取下一页
more_tweets = await tweets.next()

# 获取上一页
prev_tweets = await tweets.previous()
```
